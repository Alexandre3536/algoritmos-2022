from random import randint

def preencher_valores(gabarito,valor,repeticoes):  
  for i in range(repeticoes):    
    preenchido = True
    while(preenchido == True):
      x = randint(0,7)
      y = randint(0,7)      
      if (gabarito[x][y] == 0):
        gabarito[x][y] = valor
        preenchido = False
  return gabarito
